interface callback-types {
  resource callback {
    call: func() -> u32
  }

  export callback-types

package foo:bar

interface callback-types {

  import register-callback: func(callback: callback)
  export run: func()

Then wit-bindgen somehow assumes that callback is a type import from the host. Is this intended?
