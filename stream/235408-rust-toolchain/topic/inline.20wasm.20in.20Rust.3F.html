#[no_mangle]
pub fn inline() -> u32 {
    let x = 13;
    let y = 37;
    let mut z = 0;
    unsafe {
        asm!(
            "local.get {x}",
            "local.get {y}",
            "i32.add",
            "local.set {z}",
            x = in(local) x,
            y = in(local) y,
            z = out(local) z,
        );
    }
    z
}
