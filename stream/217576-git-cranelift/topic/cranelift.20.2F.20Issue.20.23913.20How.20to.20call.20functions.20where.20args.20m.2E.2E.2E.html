For example:

#[derive(C)]
struct S {
    a: u64,
    b: u64,
    c: u64,
    d: u64,
    e: u64,
}

extern "C" {
    fn take_a(a: A);
}

When calling take_a the argument must be put in the input argument area of the stack.
