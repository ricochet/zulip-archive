Thank you! This makes a lot of sense. I added a simple implementation, works fine

pub fn base(&self) -> *const u8 {
        match self {
            GuestMemory::Unshared(slice) => slice.as_ptr(),
            GuestMemory::Shared(slice) => slice.as_ptr() as *const u8,
        }
    }
